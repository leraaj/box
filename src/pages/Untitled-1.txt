 <section
      className="my-container d-flex flex-column"
      style={{ minHeight: "100vh" }}>
      <Navbar />

      <div
        className="container-fluid flex-grow-1 d-flex flex-column"
        style={{ backgroundColor: "#18191a" }}>
        <div className="col d-flex justify-content-end align-items-center mb-2 text-light">
          <strong>Total: {filteredMessages.length}</strong>
        </div>

        {/* Messages */}
        <div className="messages-wrapper flex-grow-1 overflow-auto p-2 border rounded">
          {loading ? (
            <div
              className="d-flex justify-content-center align-items-center"
              style={{ height: "100%", width: "100%" }}>
              <span
                className="spinner-grow spinner-grow-sm"
                style={{ width: "3rem", height: "3rem" }}
                aria-hidden="true"></span>
            </div>
          ) : filteredMessages.length === 0 ? (
            <p className="text-center text-light">No messages found.</p>
          ) : (
            filteredMessages.map((m, i) => (
              <div
                key={i}
                className={`message ${
                  m.type == 1 ? "me" : m.type == 2 ? "you" : ""
                }`}>
                <div className="message-body">{m.body}</div>
                <div className="message-datetime">
                  {(() => {
                    const d = new Date(m.date);
                    const month = d.getMonth() + 1;
                    const day = d.getDate();
                    const year = d.getFullYear();
                    let hours = d.getHours();
                    const minutes = d.getMinutes().toString().padStart(2, "0");
                    const ampm = hours >= 12 ? "PM" : "AM";
                    hours = hours % 12 || 12;
                    return `${month}/${day}/${year} - ${hours}:${minutes} ${ampm}`;
                  })()}
                </div>
              </div>
            ))
          )}
          <div ref={scrollRef}></div>
        </div>

        {/* Filters */}
        <div className="px-2 pt-2 row justify-content-end align-items-center">
          <div className="col-md-4 col-12 mb-2">
            <input
              type="search"
              className="form-control"
              placeholder="Search messages..."
              value={search}
              onChange={(e) => setSearch(e.target.value)}
            />
          </div>
          <div className="col-md-4 col-12 mb-2">
            <select
              className="form-select"
              value={filter}
              onChange={(e) => setFilter(e.target.value)}>
              <option value="">-- All Messages --</option>
              {monthYearOptions.map((opt, i) => (
                <option key={i} value={opt}>
                  {opt}
                </option>
              ))}
            </select>
          </div>
        </div>
      </div>
    </section>
